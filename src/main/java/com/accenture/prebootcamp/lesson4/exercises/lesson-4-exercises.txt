         public static void main (String[] args){

              System.out.println(result);

              System.out.println(result);

              result = result * 2; // result is now 4
              System.out.println(result);

              result = result / 2; // result is now 2
              System.out.println(result);

              result = result + 8; // result is now 10

              System.out.println(result);
         }
    }
